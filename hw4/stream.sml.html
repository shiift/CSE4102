 11 (*********************************************************************
 12  * Takes two streams and returns a stream that alternates between
 13  * the values generated by each stream.
 14  ********************************************************************)
 15 fun interleave s1 Empty = s1
 16   | interleave Empty s2 = s2
 17   | interleave s1 s2 =
 18       let fun rest () = interleave s2 (tail s1)
 19       in Stream (head s1, rest)
 20       end
 21   ;
 22 
 23 (*********************************************************************
 24  * Takes two streams and returns a stream that returns the elements
 25  * of both streams joined as tuples.
 26  ********************************************************************)
 27 fun zip s1 Empty = Empty
 28   | zip Empty s2 = Empty
 29   | zip s1 s2 =
 30       let fun rest () = zip (tail s1) (tail s2)
 31       in Stream ((head s1, head s2), rest)
 32       end
 33   ;

 35 (*********************************************************************
 36  * Takes an integer, i, and generates a stream that returns integers
 37  * that are multiples of i starting from 0.
 38  ********************************************************************)
 39 fun countBy i = generate 0 (inc i);
 40 
 41 (*********************************************************************
 42  * Takes an integer, i, and a stream and returns the ith value of the
 43  * stream.
 44  ********************************************************************)
 45 fun nth i s = head (drop (i-1) s);
 46 
 47 (*********************************************************************
 48  * Takes an integer, a, and generates a stream that always returns a.
 49  ********************************************************************)
 50 fun const a = generate a (inc 0);
 51 
 52 (*********************************************************************
 53  * Converts a list to a stream. Takes a list and returns a stream
 54  * that returns the elements of the list until the end of the list.
 55  ********************************************************************)
 56 fun listToStream [] = Empty
 57   | listToStream (head::tail) = Stream (head, (fn () => listToStream tail))
 58   ;
 59 
 60 (*********************************************************************
 61  * Converts a stream to a list. Takes a stream and returns a list
 62  * that contains the elements of the stream until the end of the
 63  * stream. This function diverges if the stream is unbounded.
 64  ********************************************************************)
 65 fun streamToList Empty = []
 66   | streamToList s = (head s)::(streamToList (tail s))
 67   ;
 68 
 69 (*********************************************************************
 70  * Takes an integer, i, and a stream and returns a stream that will
 71  * return the first i elements of s.
 72  ********************************************************************)
 73 fun first 0 s = Empty
 74   | first i s = Stream (head s, (fn () => first (i-1) (tail s)))
 75   ;

 77 (*********************************************************************
 78  * Takes integers, i and j and returns a stream that returns the
 79  * integers from i to j (inclusive).
 80  ********************************************************************)
 81 fun fromTo i j = first (j-i+1) (generate i (inc 1));
 82 
 83 (*********************************************************************
 84  * Takes two streams and joins them end-to-end. Returns a stream that
 85  * will return elements from the first stream until Empty then returns
 86  * the elements of the second stream.
 87  ********************************************************************)
 88 fun join Empty s2 = s2
 89   | join s1 Empty = s1
 90   | join s1 s2 = Stream (head s1, (fn () => join (tail s1) s2))
 91   ;
 92 
 93 (*********************************************************************
 94  * Helper function that behaves similarly to join, except that at the
 95  * end of the second stream, it loops the stream again.
 96  ********************************************************************)
 97 fun loop' Empty s2 = loop' s2 s2
 98   | loop' s1 s2 = Stream (head s1, (fn () => loop' (tail s1) s2))
 99 (*********************************************************************
100  * Takes a stream and loops the stream indefinitely.
101  ********************************************************************)
102 and loop s = loop' s s
103   ;

105 (*********************************************************************
106  * Takes integers, a, c, m, and seed and returns a stream that returns
107  * the values of the LCG rand function.
108  ********************************************************************)
109 fun rand a c m seed =
110       let val sd = (a * seed + c) mod m
111       in Stream (sd, (fn () => rand a c m sd))
112       end
113   ;
