 7 
 8 exception NameNotBound of string;
 9 
10 type 'a Env = string -> 'a;
11 
12 (* env_new: creates a new blank environment *)
13 fun env_new () : 'a Env = fn (name:string) => raise NameNotBound name;
14 
15 (* env_bind: binds a name and value to the environment *)
16 fun env_bind (env: 'a Env) (name:string) (value): 'a Env =
17   fn (input:string) =>
18     if input = name then value
19     else env input
20 
21 (* env_lookup: retreives the value of a binding in the environment *)
22 fun env_lookup (env: 'a Env) (name) = env name
