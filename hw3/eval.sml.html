 11 exception InvalidEvaluationError of string;
 12 exception InvalidCondError of string;
 13 exception InvalidParameterError of string;
 14 exception InvalidApplicationError of string;

 16 (* Expression Datatype with Constructors *)
 17 datatype Expr = Bool of bool (* boolean *)
 18               | Int of int (* integer *)
 19               | Add of Expr * Expr (* adds two expressions on evaluation *)
 20               | Cond of Expr * Expr * Expr (* given in the form (cond, conseq,
 21                 alt *)
 22               | Sym of string (* symbol that is used to assign names to values
 23                 in the environment *)
 24               | Let of (string * Expr) list * Expr (* contains a list of (id,
 25                 value) tuples with an expression to evaluate *)
 26               | Def of string * Expr (* definition that makes a new binding in
 27                 the environment when evaluated *)
 28               | Fun of Expr * Expr (* function, when evaluated, returns a
 29                 closure *)
 30               | App of Expr * Expr (* application of a function, when evaluated,
 31                 runs the function *)
 32               | Seq of Expr list (* a list of expressions *)
 33               | Disp of Expr (* displays the string format of the expression *)
 34               | Clos of Expr * Expr * Expr Env (* the function containing the
 35               environment where the function was defined *)
 36               | Nothing; (* the null datatype *)
 37 
 38 (* Expression to String Function *)
 39 
 40 fun e2s (Bool b) = "Bool(" ^ (Bool.toString b) ^ ")"
 41   | e2s (Int i)  = "Int(" ^ (Int.toString i) ^ ")"
 42   | e2s (Add (x, y)) = "Add(" ^ (e2s x) ^ "," ^ (e2s y) ^ ")"
 43   | e2s (Cond (cond, conseq, alt)) = "Cond(" ^ (e2s cond) ^ "," ^ (e2s conseq) ^
 44     "," ^ (e2s alt) ^ ")"
 45   | e2s (Sym s) = "Sym(" ^ s ^ ")"
 46   (* Let needs a helper that displays each id:value pair in the list *)
 47   | e2s (Let (l, e)) = "Let([" ^ (e2s_let_helper l) ^ "]," ^ (e2s e) ^ ")"
 48   | e2s (Def (s, e)) = "Def(" ^ s ^ "," ^ (e2s e) ^ ")"
 49   | e2s (Fun (e, x)) = "Fun(" ^ (e2s e) ^ "," ^ (e2s x) ^ ")"
 50   | e2s (App (f, p)) = "App(" ^ (e2s f) ^ "," ^ (e2s p) ^ ")"
 51   (* Seq needs a helper that displays each expression in the list *)
 52   | e2s (Seq s) = "Seq([" ^ (e2s_seq_helper s) ^ "])"
 53   | e2s (Disp d) = "Disp(" ^ (e2s d) ^ ")"
 54   | e2s (Clos (e, x, _)) = "Clos(" ^ (e2s e) ^ "," ^ (e2s x) ^ ")"
 55   | e2s (Nothing) = "(null)"
 56     (* Let helper pattern matches on possible formats and recursively displays
 57     * the id:value pairs through the end of the list *)
 58     and e2s_let_helper ([]) = ""
 59       | e2s_let_helper ((id:string, e:Expr)::[]) = (id ^ ":" ^ (e2s e))
 60       | e2s_let_helper ((id:string, e:Expr)::tail) =
 61             (id ^ ":" ^ (e2s e) ^ "," ^ (e2s_let_helper tail))
 62     (* Seq helper pattern matches on possible formats and recursively displays
 63     * the expressions through the end of the list *)
 64     and e2s_seq_helper ([]) = ""
 65       | e2s_seq_helper (head::[]) = (e2s head)
 66       | e2s_seq_helper (head::tail) =
 67             (e2s head) ^ "," ^ (e2s_seq_helper tail);

 69 (* Evaluation Function *)
 70 
 71 fun eval (env:Expr Env) (Sym s) = (env,(env s))
 72   (* Add: evaluates the two input expressions, then adds the results *)
 73   | eval env (Add (x, y)) =
 74     (case ((eval env x), (eval env y)) of
 75           ((_,Int c),(_,Int d)) => (env, Int (c + d))
 76         | (c, d) => raise InvalidEvaluationError "Add")
 77   (* Cond: evaluates the cond, if the cond returns true, then evaluate cons,
 78   * else evaluate the alt *)
 79   | eval env (Cond (cond,cons,alt)) =
 80     (case (eval env cond) of
 81           (_,Bool b) => if b then (eval env cons)
 82                         else (eval env alt)
 83         | (_) => raise InvalidCondError "Cond Requires Bool")
 84   (* Let: use the helper function to setup the environment with the bindings
 85   * given, then evaluate the expression and return the result with the original
 86   * environment *)
 87   | eval env (Let (b, e)) =
 88     let
 89         val env2 = (eval_let_helper env b)
 90         val (_,v) = (eval env2 e)
 91     in
 92         (env, v)
 93     end
 94   (* Def: bind the definition to the environment and return it, return Nothing
 95   * as the value *)
 96   | eval env (Def (s, e)) = ((env_bind env s e), Nothing)
 97   (* Seq: use the helper to evaluate each expression in the sequence *)
 98   | eval env (Seq s) = (eval_seq_helper env s)
 99   (* Disp: print out the e2s of the expression and return Nothing *)
100   | eval env (Disp d) = ((print ((e2s d) ^ "\n")); (env, Nothing))
101   (* Fun: create and return a closure containing the environment at the time of
102   * function creation *)
103   | eval env (Fun (e, x)) = (env, Clos (e, x, env))
104   (* App: find the closure for the function, run the closure with the parameters
105   * provided, then return the return value of the function with the original
106   * environment *)
107   | eval env (App (func, p)) =
108     (case (eval env func) of
109           (_, Clos (arg, steps, envc)) =>
110             (case arg of
111                   (Sym s) =>
112                     let
113                       val (_, return) = (eval (env_bind envc s p) steps)
114                     in
115                       (env, return)
116                     end
117                 | (_) => raise InvalidParameterError "Parameter must be a Symbol")
118         | (_) => raise InvalidApplicationError "Application must be applied to a Fun")
119   (* catch all for other expression types *)
120   | eval env expr = (env, expr)A
121     (* Let helper binds all of the id:value pairs in the let and returns the new
122     * resulting environment *)
123     and eval_let_helper env [] = env
124       | eval_let_helper env ((id, v)::tail) =
125         (eval_let_helper (env_bind env id v) tail)
126     (* Seq helper recursively evaluates all of the expressions in the sequence
127     * and returns the result of the final expression in the sequence *)
128     and eval_seq_helper env [] = (env, Nothing) (* edge case catch *)
129       | eval_seq_helper env (head::[]) = (eval env head)
130       | eval_seq_helper env (expr::tail) =
131         let
132           val (e,_) = (eval env expr)
133         in
134           (eval_seq_helper e tail)
135         end;
